create table Theater(
    theaterID integer primary key generated always as identity(Start with 1, Increment by 1),
    theaterName varchar(50),
    theaterAddress varchar(80),
    theaterZip char(5)
);

create table Movie(
    movieID integer primary key generated always as identity(Start with 1, Increment by 1),
    movieName varchar(50),
    movieDesc varchar(1000),
    movieRating varchar(5)
);

create table MovieTimes(
    MT_ID integer primary key generated always as identity(Start with 1, Increment by 1),
    MT_theaterID integer,
    MT_movieID integer,
    MT_showtime varchar(8),
    foreign key(MT_theaterID) references Theater(theaterID),
    foreign key(MT_movieID) references Movie(movieID)
);
